1. شبكات بايثون والـ Sockets

كود بايثون تفاعلي كامل لإجراء اتصال TCP Socket مخصص بالهدف:

import socket

target_host = "10.10.10.25"
target_port = 9999

client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client.connect((target_host, target_port))
client.send(b"GET / HTTP/1.1\r\nHost: 10.10.10.25\r\n\r\n")
response = client.recv(4096)
print(response.decode())
